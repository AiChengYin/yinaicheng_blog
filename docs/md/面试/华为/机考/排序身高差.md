# 排序身高差

```java
import java.util.ArrayList;
import java.util.List;
import java.util.Scanner;
/**
 * 小明今年升学到了小学1年纪，来到新班级后，发现其他小朋友身高参差不齐，然后就想基于各小朋友和自己的身高差，对他们进行排序，请帮他实现排序
 * 输入描述，第一行为正整数 h和n，0<h<200 为小明的身高，0<n<50 为新班级其他小朋友个数
 * 第二行为n各正整数，h1 ~ hn分别是其他小朋友的身高，取值范围0<hi<200，且n个正整数各不相同
 * 输出描述：输出排序结果，各正整数以空格分割，和小明身高差绝对值最小的小朋友排在前面，和小明身高差绝对值最大的小朋友排在后面，如果两个小朋友和小明身高差一样，则个子较小的小朋友排在前面
 * @author: yinaicheng
 */
public class SortHeightAbsoluteValue {

    public static void main(String[] args) {
        /*读取输入数据*/
        Scanner scanner=new Scanner(System.in);
        /*小明身高+其他小朋友个数*/
        String[] xiaoMingHeigthAndOtherStudentNumbers=scanner.nextLine().split(" ");
        int xiaoMingHeight=Integer.parseInt(xiaoMingHeigthAndOtherStudentNumbers[0]);
        /*其他小朋友的身高数组（字符串）*/
        String[] otherStudentHeightArray=scanner.nextLine().split(" ");
        /*其他小朋友的身高列表（整数）*/
        List<Integer> otherStudentHeightList=new ArrayList<>();
        for(String otherStudentHeight:otherStudentHeightArray){
            otherStudentHeightList.add(Integer.parseInt(otherStudentHeight));
        }
        /*对数组进行排序，先按照绝对值排序，如果绝对值相同，就按个头排序*/
        otherStudentHeightList.sort((o1, o2) -> {
            int o1AbsoluteValue=getAbsoluteValue(o1-xiaoMingHeight);
            int o2AbsoluteValue=getAbsoluteValue(o2-xiaoMingHeight);
            /*如果两个小朋友和小明身高差一样，则个子较小的小朋友排在前面*/
            if(o1AbsoluteValue==o2AbsoluteValue){
                return o1-o2;
            }
            else{
                return o1AbsoluteValue-o2AbsoluteValue;
            }
        });
        /*展示*/
        StringBuilder stringBuilder=new StringBuilder();
        for(Integer otherStudentHeight:otherStudentHeightList){
            stringBuilder.append(otherStudentHeight).append(" ");
        }
        System.out.println(stringBuilder.toString().trim());
        scanner.close();
    }

    private static int getAbsoluteValue(int value){
        return value<0?-value:value;
    }

}
```
