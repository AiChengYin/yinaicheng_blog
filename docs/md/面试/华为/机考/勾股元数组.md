# 勾股元数组

```java
import java.text.MessageFormat;
import java.util.Scanner;
/**
 * 如果三个正整数A B C ,A²+B²=C²则为勾股数，如果ABC之间两两互质，即A与B A与C B与C均互质没有公约数，则称其为勾股数元组
 * 请求出给定n m 范围内所有的勾股数元组
 * 输入描述：起始范围 1<n<10000、n<m<10000
 * 输出目描述：abc 保证a<b<c输出格式  a b c
 * 多组勾股数元组 按照a升序b升序 c升序的排序方式输出。
 * 给定范围内，找不到勾股数元组时，输出  Na
 * 勾股数元组
 * @author: yinaicheng
 */
public class PythagoreanTuple {

    public static void main(String[] args) {
        Scanner scanner=new Scanner(System.in);
        int count=0;
        /*输入的n值*/
        int n=scanner.nextInt();
        /*输入的m值*/
        int m=scanner.nextInt();
        /*校验输入值*/
        if((n>1 && n<10000) && (m>n && m<10000)){
            /*暴力破解，c>b>a，一个个试*/
            for(int a=n;a<m;a++){
                for(int b=n+1;b<m;b++){
                    for(int c=n+2;c<m;c++){
                        if (a<b && b<c && (c*c==a*a+b*b) && getMaxCommonDivisor(a,b)==1 && getMaxCommonDivisor(a,c)==1 && getMaxCommonDivisor(b,c)==1) {
                            /*计数*/
                            count++;
                            System.out.println(MessageFormat.format("{0} {1} {2}",a,b,c));
                        }
                    }
                }
            }
        }
        /*找不到勾股数元组时，输出Na*/
        if(0==count){
            System.out.println("Na");
        }
    }

    /**
     * 求最大公约数
     */
    private static  int getMaxCommonDivisor(int i,int j){
        /*判断边界情况*/
        if(i==0 || j==0){
            return 1;
        }
        /*取余*/
        return i%j==0?j:getMaxCommonDivisor(j,i%j);
    }
}
```
