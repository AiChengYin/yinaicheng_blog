# 安装Docker

# 环境要求及版本

系统环境：Centos Linux release 7

# 安装步骤

使用yum安装
执行以下命令安装依赖包
sudo yum install -y yum-utils
鉴于国内网络环境，使用国内源
sudo yum-config-manager --add-repo https://mirrors.aliyun.com/docker-ce/linux/centos/docker-ce.repo
sudo sed -i 's/download.docker.com/mirrors.aliyun.com\/docker-ce/g' /etc/yum.repos.d/docker-ce.repo
安装Docker
更新 yum 软件源缓存，并安装 docker-ce
sudo yum install docker-ce docker-ce-cli containerd.io

# Docker设置国内镜像源

手动先进入etc目录下:

cd /etc/
1

手动创docker建文件夹并且进入docker文件夹：

mkdir docker/
cd docker/
1
2

创建daemon.json文件并且进入编辑：

vi /etc/docker/daemon.json
1

进入vi图形界面，增添：

{
    "registry-mirrors": ["http://hub-mirror.c.163.com"]
}
或者自己的镜像加速器，如
{
    "registry-mirrors": ["https://0d3bq9mn.mirror.aliyuncs.com"]
}

# docker的/var/lib/docker目录迁移

# 首先需要停止docker服务

systemctl stop docker
1

# 通过命令df -h去选择磁盘一个大的空间

# 创建新的docker目录

# 参数-p确保目录名称存在，如果目录不存在就创建
mkdir -p /data/docker/lib
1
2

# 开始迁移

# 先确认是否安装rsync
# -a:归档模式，表示递归传输并且保持文件属性
# -v:显示rsync过程中详细信息
# -p:显示文件传输的进度信息
# -z:传输时进行压缩提高效率
rsync -avzP /var/lib/docker /data/docker/lib
